Example OSC Client (Python)

import socket
import json

# OSC client setup
client = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)

# Send OSC message
def send_osc(address, *args):
    message = json.dumps({"address": address, "args": args})
    client.sendto(message.encode(), ("127.0.0.1", 12000))

# Example: Start recording
send_osc("/startRecording")

# Example: Stop recording
send_osc("/stopRecording")
